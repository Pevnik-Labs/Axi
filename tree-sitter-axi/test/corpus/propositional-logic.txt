theorem resolution : (P \/ Q) /\ (~ P \/ R) --> Q \/ R
proof
  assume (both pq npr)
  cases pq ?pr ?qq
  lemma qq by
    assume q
    or-left q
  lemma pr by
    cases npr
    . assume np
      absurd
      apply contradiction np p
    . assume r
      or-right r
qed

theorem resolution' : (P \/ Q) /\ (~ P \/ R) --> Q \/ R

  assume (both np npr)
  cases pq (assume p in ?pr) (assume q in ?qq)

    . assume np

      apply contradiction np p
    . assume r
      or-right r
qed
