\section{Complete and Easy Quantitative Contextual Typing}

\begin{frame}{Judgements}

\begin{itemize}
  \item Typing: $\infer{\Gamma}{e}{A}{\Gamma'}$ -- in context $\Gamma$, term $e$ infers type $A$, with output context $\Gamma'$
  \item Matching: $\matching{\Gamma}{A}{B}{\Gamma'}$ -- in context $\Gamma$, type $A$ matches (output) type $B$, with output context $\Gamma'$
  \item Subtyping: $\subtyping{\Gamma}{r}{A}{B}{s}{\Gamma'}$ -- in context $\Gamma$, $s$ copies of subtype $A$ are needed to produce $r$ copies of supertype $B$, with output context $\Gamma'$.
  \item Sub-instantiation: $\subinst{\Gamma}{r}{\evar{a}}{A}{s}{\Gamma'}$ -- in context $\Gamma$, instantiate the existential variable $\evar{a}$ to a subtype of $\Box[r]{A}$, lacking $s$, returning context $\Gamma'$
  \item Super-instantiation: $\superinst{\Gamma}{r}{A}{\evar{a}}{s}{\Gamma'}$ -- in context $\Gamma$, instantiate the existential variable $\evar{a}$, so that $A <: \Box[r]{\evar{a}}$, lacking $s$, returning context $\Gamma$
\end{itemize}

\end{frame}
